106@execute_cli.command() 

107@__global_test_options 

108@click.argument("cloud-bucket", type=click.STRING) 

109@click.argument("local-file-pattern", type=click.STRING, nargs=UNLIMITED_ARGS) 

110def push(fail_fast, local_file_pattern, cloud_bucket): 

111 """Push files from local machine to the cloud bucket.""" 

112 patterns = list(local_file_pattern) 

113 cloud_map_list = [] 

114 for pattern in patterns: 

115 pattern_expansion = glob.glob(pattern, recursive=False) 

116 

117 # Only try to upload files, exclude any directories 

118 pattern_expansion = list( 

119 filter(lambda path: os.path.isfile(path), pattern_expansion) 

120 ) 

121 

122 # Either the pattern expansion is a list of files, or it's a file itself 

123 if len(pattern_expansion) == 0: 

124 __add_if_file_exists(cloud_map_list, pattern) 

125 else: 

126 for filepath in pattern_expansion: 

127 __add_if_file_exists(cloud_map_list, filepath) 

128 

129 if len(cloud_map_list) > 0: 

130 pbar = tqdm( 

131 total=len(cloud_map_list), 

132 desc="Uploading", 

133 unit=PROGRESS_BAR_UNITS, 

134 colour=PROGRESS_BAR_COLOR, 

135 ) 

136 

137 file_broker.upload_files( 

138 cloud_bucket, 

139 cloud_map_list, 

140 lambda bucket_name, cloud_key, file_path, succeeded: __update_pbar_with_filenames( 

141 "upload", fail_fast, pbar, file_path, succeeded 

142 ), 

143 ) 

144 

145 pbar.close() 

146 else: 

147 print("Nothing to push.")
